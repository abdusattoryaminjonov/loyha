#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <DHT.h>

#define DHTPIN 2          // DHT22 DATA pin ulangan pin
#define DHTTYPE DHT22     // Sensor turi

DHT dht(DHTPIN, DHTTYPE);
LiquidCrystal_I2C lcd(0x27, 16, 2);  // 0x27 - I2C manzili, 16x2 ekran

void setup() {
  lcd.begin(16, 2);     // LCD o'lchamlari ko'rsatilmoqda
  lcd.backlight();      // Orqa yoritishni yoqish
  dht.begin();          // DHT sensorni ishga tushirish
  lcd.setCursor(0, 0);
  lcd.print("Tizim yuklanmoqda");
  delay(2000);
  lcd.clear();
}

void loop() {
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();

  if (isnan(temp) || isnan(hum)) {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Sensor xatosi");
    delay(2000);
    return;
  }

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("T: ");
  lcd.print(temp, 1);
  lcd.print((char)223);  // Â° belgisi
  lcd.print("C");

  lcd.setCursor(0, 1);
  lcd.print("H: ");
  lcd.print(hum, 1);
  lcd.print("%");

  // Harorat nazorati
  if (temp > 30) {
    lcd.setCursor(10, 1);
    lcd.print("YUQORI");
  } else if (temp < 10) {
    lcd.setCursor(10, 1);
    lcd.print("PAST");
  }

  delay(2000);
}
